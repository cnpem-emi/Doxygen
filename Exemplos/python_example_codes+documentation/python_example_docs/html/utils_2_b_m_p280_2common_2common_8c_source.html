66int8_t bmp_init(struct sensor_data* sensor, uint8_t addr) {
67 int8_t rslt;
68
69 if (configure_mux()) {
70 syslog(LOG_CRIT, "Failed to configure demux switching.\n");
71 exit(1);
72 }
73
74 direct_mux(sensor->id.mux_id);
75
76 rslt = bmp2_interface_selection(sensor, BMP2_I2C_INTF, addr);
77 if (rslt)
78 return rslt;
79
80 rslt = bmp2_get_config(&sensor->config, &sensor->dev);
81 if (rslt)
82 return rslt;
83
84 sensor->config.os_mode = BMP2_OS_MODE_ULTRA_HIGH_RESOLUTION;
85 sensor->config.filter = BMP2_FILTER_OFF;
86 sensor->config.odr = BMP2_ODR_250_MS;
87
88 rslt = bmp2_set_config(&sensor->config, &sensor->dev);
89 if (rslt)
90 return rslt;
91
92 rslt = bmp2_set_power_mode(BMP2_POWERMODE_FORCED, &sensor->config, &sensor->dev);
93 if (rslt)
94 return rslt;
95
96 return bmp2_init(&sensor->dev);
97}

102int8_t bmp_read(struct sensor_data* sensor) {
103 struct bmp2_status status;
104 int8_t rslt;
105
106 direct_mux(sensor->id.mux_id);
107
108 rslt = bmp2_get_status(&status, &sensor->dev);
109 for (int i; i < 20; i++) {
110 if (status.measuring == BMP2_MEAS_DONE) {
111 sensor->dev.delay_us(1000, sensor->dev.intf_ptr);
112 rslt = bmp2_get_sensor_data(&sensor->data, &sensor->dev);
113 sensor->data.pressure /= 100;
114
115 break;
116 }
117 bmp2_get_status(&status, &sensor->dev);
118 }
119
120 return rslt;
121}
