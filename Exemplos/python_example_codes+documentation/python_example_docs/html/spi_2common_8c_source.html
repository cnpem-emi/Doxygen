45int mmio_get_gpio(gpio_t* gpio) {
46 // Validate input parameters.
47 int base = gpio->pin / 32;
48 int number = gpio->pin % 32;
49
50 if (gpio == NULL)
51 return MMIO_ERROR_ARGUMENT;
52 if (base < 0 || base > 3)
53 return MMIO_ERROR_ARGUMENT;
54 if (number < 0 || number > 31)
55 return MMIO_ERROR_ARGUMENT;
56 if (gpio_base[base] == NULL) {
57 int mfd = open("/dev/mem", O_RDWR | O_SYNC);
58 if (mfd == -1)
59 return MMIO_ERROR_DEVMEM;
60
61 gpio_base[base] = (uint32_t*)mmap(NULL, GPIO_LENGTH, PROT_READ | PROT_WRITE, MAP_SHARED, mfd,
62 gpio_addresses[base]);
63 if (gpio_base[base] == MAP_FAILED) {
64 gpio_base[base] = NULL;
65 return MMIO_ERROR_MMAP;
66 }
67 }
68
69 memset(gpio, 0, sizeof(*gpio));
70 gpio->base = gpio_base[base];
71 gpio->number = number;
72 return MMIO_SUCCESS;
73}

166int select_module(int address, int module) {
167 unsigned long msg;
168 int parity = calculate_parity(address);

171 msg = (msg << 3) | module;
