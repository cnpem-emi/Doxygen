160int lcd_set_cursor(int row, int col) {
161 if (row < 0 || row > LCD_HEIGHT / 8 + 1 || col < 0 || col > LCD_WIDTH / 6 + 1)
162 return 1;
163
164 char rowb[1] = {(64 | (row))};
165 char colb[1] = {(128 | (col)*6)};
166
167 lcd_write(0, rowb, 1);
168 lcd_write(0, colb, 1);
169
170 return 0;
171}

188void lcd_print(char* strn) {
189 int slen = strlen(strn);
190 for (int i = 0; i < slen; i++) {

208void lcd_line(int x, int y, int len, int thickness) {
209 lcd_set_cursor(y / 8, x / 6);
210 int mod_x = x % 6;
211 int mod_y = y % 8;
212
213 int fix_len = len + 1;
214
215 char c;
216 char full_char[fix_len];
217
218 c = 0x01 << mod_y;
219
220 if (thickness > 1) {
221 for (int i = 1; i < thickness; i++)
222 c |= c << 1;
223 }
224
225 for (int i = 0; i < fix_len; i++)
226 full_char[i] = i >= mod_x ? c : 0;
227
228 lcd_write(1, full_char, fix_len);
229}

235void lcd_init(char* vop, char* bias) {
236 fd = spi_open(device, &spi_mode, &bits, &speed);
237
238 mmio_get_gpio(&rst);
239 mmio_get_gpio(&bl);
240 mmio_get_gpio(&dc);
241
242 mmio_set_output(rst);
243 mmio_set_output(bl);
244 mmio_set_output(dc);
245
246 mmio_set_low(rst);
247 nanosleep((const struct timespec[]){{0, 500000000L}}, NULL);
248 mmio_set_high(rst);
249 nanosleep((const struct timespec[]){{0, 500000000L}}, NULL);
250
251 lcd_write(0, "\x21", 1); // Ext. cmd set
252 lcd_write(0, vop, 1); // LCD contrast (VOP)
253 lcd_write(0, "\x04", 1); // Temperature coefficient
254 lcd_write(0, bias, 1); // Voltage Bias
255
256 lcd_write(0, "\x20", 1); // Normal cmd set
257 lcd_write(0, "\x0c", 1); // Normal display (0x0d for inverted)
258
259 lcd_clear();
260
261 lcd_set_backlight(0);
262 lcd_set_cursor(0, 0);
263}
