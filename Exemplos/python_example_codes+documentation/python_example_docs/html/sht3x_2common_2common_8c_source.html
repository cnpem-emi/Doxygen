62uint8_t sensirion_common_generate_crc(const uint8_t* data, uint16_t count) {
63 uint16_t current_byte;
64 uint8_t crc = CRC8_INIT;
65 uint8_t crc_bit;
66
67 /* calculates 8-Bit checksum with given polynomial */
68 for (current_byte = 0; current_byte < count; ++current_byte) {
69 crc ^= (data[current_byte]);
70 for (crc_bit = 8; crc_bit > 0; --crc_bit) {
71 if (crc & 0x80)
72 crc = (crc << 1) ^ CRC8_POLYNOMIAL;
73 else
74 crc = (crc << 1);
75 }
76 }
77 return crc;
78}

80int8_t sensirion_common_check_crc(const uint8_t* data, uint16_t count, uint8_t checksum) {
81 if (sensirion_common_generate_crc(data, count) != checksum)
82 return STATUS_FAIL;
83 return NO_ERROR;
84}

97uint16_t sensirion_fill_cmd_send_buf(uint8_t* buf,
98 uint16_t cmd,
99 const uint16_t* args,
100 uint8_t num_args) {
101 uint8_t crc;
102 uint8_t i;
103 uint16_t idx = 0;
104
105 buf[idx++] = (uint8_t)((cmd & 0xFF00) >> 8);
106 buf[idx++] = (uint8_t)((cmd & 0x00FF) >> 0);
107
108 for (i = 0; i < num_args; ++i) {
109 buf[idx++] = (uint8_t)((args[i] & 0xFF00) >> 8);
110 buf[idx++] = (uint8_t)((args[i] & 0x00FF) >> 0);
111
112 crc = sensirion_common_generate_crc((uint8_t*)&buf[idx - 2], SENSIRION_WORD_SIZE);
113 buf[idx++] = crc;
114 }
115 return idx;
116}

118int8_t sensirion_i2c_read_words_as_bytes(struct sht3x_sensor_data* sensor,
119 uint8_t* data,
120 uint16_t num_words) {
121 int8_t ret;
122 uint16_t i, j;
123 uint32_t size = num_words * (SENSIRION_WORD_SIZE + CRC8_LEN);
124 uint16_t word_buf[SENSIRION_MAX_BUFFER_WORDS];
125 uint8_t* const buf8 = (uint8_t*)word_buf;
126
127 direct_mux(sensor->id.mux_id);
128
129 if (sensor->id.ext_mux_id >= 0)
130 direct_ext_mux(sensor->id.ext_mux_id);
131
132 delay_us(1000, NULL);
133 ret = i2c_read(0, buf8, size, &sensor->id);
134 if (ret != NO_ERROR)
135 return ret;
136
137 /* check the CRC for each word */
138 for (i = 0, j = 0; i < size; i += SENSIRION_WORD_SIZE + CRC8_LEN) {
139 ret = sensirion_common_check_crc(&buf8[i], SENSIRION_WORD_SIZE, buf8[i + SENSIRION_WORD_SIZE]);
140 if (ret != NO_ERROR)
141 return ret;
142
143 data[j++] = buf8[i];
144 data[j++] = buf8[i + 1];
145 }
146
147 return NO_ERROR;
148}

150int8_t sensirion_i2c_read_words(struct sht3x_sensor_data* sensor,
151 uint16_t* data_words,
152 uint16_t num_words) {
153 int8_t ret;
154 uint8_t i;
155 const uint8_t* word_bytes;
156
157 ret = sensirion_i2c_read_words_as_bytes(sensor, (uint8_t*)data_words, num_words);
158 if (ret != NO_ERROR)
159 return ret;
160
161 for (i = 0; i < num_words; ++i) {
162 word_bytes = (uint8_t*)&data_words[i];
163 data_words[i] = ((uint16_t)word_bytes[0] << 8) | word_bytes[1];
164 }
165
166 return NO_ERROR;
167}
