37/*
38 * ======== PRU INTC Map ========
39 *
40 * Define the INTC mapping for interrupts going to the ICSS / ICSSG:
41 * ICSS Host interrupts 0, 1
42 * ICSSG Host interrupts 0, 1, 10-19
43 *
44 * Note that INTC interrupts going to the ARM Linux host should not be defined
45 * in this file (ICSS/ICSSG Host interrupts 2-9).
46 *
47 * The INTC configuration for interrupts going to the ARM host should be defined
48 * in the device tree node of the client driver, "interrupts" property.
49 * See Documentation/devicetree/bindings/interrupt-controller/ti,pruss-intc.yaml
50 * entry #interrupt-cells for more.
51 *
52 * For example, on ICSSG:
53 *
54 * &client_driver0 {
55 * interrupt-parent = <&icssg0_intc>;
56 * interrupts = <21 2 2>, <22 3 3>;
57 * interrupt-names = "interrupt_name1", "interrupt_name2";
58 * };
59 *
60 */
61
62#include <rsc_types.h>
63#include <stddef.h>
64
65/*
66 * .pru_irq_map is used by the RemoteProc driver during initialization. However,
67 * the map is NOT used by the PRU firmware. That means DATA_SECTION and RETAIN
68 * are required to prevent the PRU compiler from optimizing out .pru_irq_map.
69 */
70#pragma DATA_SECTION(my_irq_rsc, ".pru_irq_map")
71#pragma RETAIN(my_irq_rsc)
72
73struct pru_irq_rsc my_irq_rsc = {
74 0, /* type = 0 */
75 1, /* number of system events being mapped */
76 {
77 {19, 1, 1}, /* {sysevt, channel, host interrupt} */
78 },
79};

81#endif /* _INTC_MAP_1_H_ */
