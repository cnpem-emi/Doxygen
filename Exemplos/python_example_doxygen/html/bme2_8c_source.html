412int8_t bme280_init(struct bme280_dev* dev) {
413 int8_t rslt;
414
415 /* chip id read try count */
416 uint8_t try_count = 5;
417 uint8_t chip_id = 0;
418
419 /* Check for null pointer in the device structure*/
420 rslt = null_ptr_check(dev);
421
422 /* Proceed if null check is fine */
423 if (rslt == BME280_OK) {
424 while (try_count) {
425 /* Read the chip-id of bme280 sensor */
426 rslt = bme280_get_regs(BME280_CHIP_ID_ADDR, &chip_id, 1, dev);
427
428 /* Check for chip id validity */
429 if ((rslt == BME280_OK) && (chip_id == BME280_CHIP_ID || chip_id == BMP280_CHIP_ID)) {
430 dev->chip_id = chip_id;
431
432 /* Reset the sensor */
433 rslt = bme280_soft_reset(dev);
434
435 if (rslt == BME280_OK) {
436 /* Read the calibration data */
437 rslt = get_calib_data(dev);
438 }
439
440 break;
441 }
442
443 /* Wait for 1 ms */
444 dev->delay_us(1000, dev->intf_ptr);
445 --try_count;
446 }
447
448 /* Chip id check failed */
449 if (!try_count) {
450 rslt = BME280_E_DEV_NOT_FOUND;
451 }
452 }
453
454 return rslt;
455}

460int8_t bme280_get_regs(uint8_t reg_addr, uint8_t* reg_data, uint16_t len, struct bme280_dev* dev) {
461 int8_t rslt;
462
463 /* Check for null pointer in the device structure*/
464 rslt = null_ptr_check(dev);
465
466 /* Proceed if null check is fine */
467 if ((rslt == BME280_OK) && (reg_data != NULL)) {
468 /* If interface selected is SPI */
469 if (dev->intf != BME280_I2C_INTF) {
470 reg_addr = reg_addr | 0x80;
471 }
472
473 /* Read the data */
474 dev->intf_rslt = dev->read(reg_addr, reg_data, len, dev->intf_ptr);
475
476 /* Check for communication error */
477 if (dev->intf_rslt != BME280_INTF_RET_SUCCESS) {
478 rslt = BME280_E_COMM_FAIL;
479 }
480 } else {
481 rslt = BME280_E_NULL_PTR;
482 }
483
484 return rslt;
485}

491int8_t bme280_set_regs(uint8_t* reg_addr,
492 const uint8_t* reg_data,
493 uint8_t len,
494 struct bme280_dev* dev) {
495 int8_t rslt;
496 uint8_t temp_buff[20]; /* Typically not to write more than 10 registers */
497
498 if (len > 10) {
499 len = 10;
500 }
501
502 uint16_t temp_len;
503 uint8_t reg_addr_cnt;
504
505 /* Check for null pointer in the device structure*/
506 rslt = null_ptr_check(dev);
507
508 /* Check for arguments validity */
509 if ((rslt == BME280_OK) && (reg_addr != NULL) && (reg_data != NULL)) {
510 if (len != 0) {
511 temp_buff[0] = reg_data[0];
512
513 /* If interface selected is SPI */
514 if (dev->intf != BME280_I2C_INTF) {
515 for (reg_addr_cnt = 0; reg_addr_cnt < len; reg_addr_cnt++) {
516 reg_addr[reg_addr_cnt] = reg_addr[reg_addr_cnt] & 0x7F;
517 }
518 }
519
520 /* Burst write mode */
521 if (len > 1) {
522 /* Interleave register address w.r.t data for
523 * burst write
524 */
525 interleave_reg_addr(reg_addr, temp_buff, reg_data, len);
526 temp_len = ((len * 2) - 1);
527 } else {
528 temp_len = len;
529 }
530
531 dev->intf_rslt = dev->write(reg_addr[0], temp_buff, temp_len, dev->intf_ptr);
532
533 /* Check for communication error */
534 if (dev->intf_rslt != BME280_INTF_RET_SUCCESS) {
535 rslt = BME280_E_COMM_FAIL;
536 }
537 } else {
538 rslt = BME280_E_INVALID_LEN;
539 }
540 } else {
541 rslt = BME280_E_NULL_PTR;
542 }
543
544 return rslt;
545}

576 */

578 rslt = set_filter_standby_settings(desired_settings, &dev->settings, dev);
579 }
580 }
581 }
582
583 return rslt;

663int8_t bme280_soft_reset(struct bme280_dev* dev) {
664 int8_t rslt;
665 uint8_t reg_addr = BME280_RESET_ADDR;
666 uint8_t status_reg = 0;
667 uint8_t try_run = 5;
668
669 /* 0xB6 is the soft reset command */
670 uint8_t soft_rst_cmd = BME280_SOFT_RESET_COMMAND;
671
672 /* Check for null pointer in the device structure*/
673 rslt = null_ptr_check(dev);
674
675 /* Proceed if null check is fine */
676 if (rslt == BME280_OK) {
677 /* Write the soft reset command in the sensor */
678 rslt = bme280_set_regs(&reg_addr, &soft_rst_cmd, 1, dev);
679
680 if (rslt == BME280_OK) {
681 /* If NVM not copied yet, Wait for NVM to copy */
682 do {
683 /* As per data sheet - Table 1, startup time is 2 ms. */
684 dev->delay_us(2000, dev->intf_ptr);
685 rslt = bme280_get_regs(BME280_STATUS_REG_ADDR, &status_reg, 1, dev);
686
687 } while ((rslt == BME280_OK) && (try_run--) && (status_reg & BME280_STATUS_IM_UPDATE));
688
689 if (status_reg & BME280_STATUS_IM_UPDATE) {
690 rslt = BME280_E_NVM_COPY_FAILED;
691 }
692 }
693 }
694
695 return rslt;
696}

703int8_t bme280_get_sensor_data(uint8_t sensor_comp,
704 struct bme280_data* comp_data,
705 struct bme280_dev* dev) {
706 int8_t rslt;
707
708 /* Array to store the pressure, temperature and humidity data read from
709 * the sensor
710 */
711 uint8_t reg_data[BME280_P_T_H_DATA_LEN] = {0};
712 struct bme280_uncomp_data uncomp_data = {0};
713
714 /* Check for null pointer in the device structure*/
715 rslt = null_ptr_check(dev);
716
717 if ((rslt == BME280_OK) && (comp_data != NULL)) {
718 /* Read the pressure and temperature data from the sensor */
719 rslt = bme280_get_regs(BME280_DATA_ADDR, reg_data, BME280_P_T_H_DATA_LEN, dev);
720
721 if (rslt == BME280_OK) {
722 /* Parse the read data from the sensor */
723 bme280_parse_sensor_data(reg_data, &uncomp_data);
724
725 /* Compensate the pressure and/or temperature and/or
726 * humidity data from the sensor
727 */
728 rslt = bme280_compensate_data(sensor_comp, &uncomp_data, comp_data, &dev->calib_data);
729 }
730 } else {
731 rslt = BME280_E_NULL_PTR;
732 }
733
734 return rslt;
735}

741void bme280_parse_sensor_data(const uint8_t* reg_data, struct bme280_uncomp_data* uncomp_data) {
742 /* Variables to store the sensor data */
743 uint32_t data_xlsb;
744 uint32_t data_lsb;
745 uint32_t data_msb;
746
747 /* Store the parsed register values for pressure data */
748 data_msb = (uint32_t)reg_data[0] << 12;
749 data_lsb = (uint32_t)reg_data[1] << 4;
750 data_xlsb = (uint32_t)reg_data[2] >> 4;
751 uncomp_data->pressure = data_msb | data_lsb | data_xlsb;
752
753 /* Store the parsed register values for temperature data */
754 data_msb = (uint32_t)reg_data[3] << 12;
755 data_lsb = (uint32_t)reg_data[4] << 4;
756 data_xlsb = (uint32_t)reg_data[5] >> 4;
757 uncomp_data->temperature = data_msb | data_lsb | data_xlsb;
758
759 /* Store the parsed register values for humidity data */
760 data_msb = (uint32_t)reg_data[6] << 8;
761 data_lsb = (uint32_t)reg_data[7];
762 uncomp_data->humidity = data_msb | data_lsb;
763}

770int8_t bme280_compensate_data(uint8_t sensor_comp,
771 const struct bme280_uncomp_data* uncomp_data,
772 struct bme280_data* comp_data,
773 struct bme280_calib_data* calib_data) {
774 int8_t rslt = BME280_OK;
775
776 if ((uncomp_data != NULL) && (comp_data != NULL) && (calib_data != NULL)) {
777 /* Initialize to zero */
778 comp_data->temperature = 0;
779 comp_data->pressure = 0;
780 comp_data->humidity = 0;
781
782 /* If pressure or temperature component is selected */
783 if (sensor_comp & (BME280_PRESS | BME280_TEMP | BME280_HUM)) {
784 /* Compensate the temperature data */
785 comp_data->temperature = compensate_temperature(uncomp_data, calib_data);
786 }
787
788 if (sensor_comp & BME280_PRESS) {
789 /* Compensate the pressure data */
790 comp_data->pressure = compensate_pressure(uncomp_data, calib_data);
791 }
792
793 if (sensor_comp & BME280_HUM) {
794 /* Compensate the humidity data */
795 comp_data->humidity = compensate_humidity(uncomp_data, calib_data);
796 }
797 } else {
798 rslt = BME280_E_NULL_PTR;
799 }
800
801 return rslt;
802}

809uint32_t bme280_cal_meas_delay(const struct bme280_settings* settings) {
810 uint32_t max_delay;
811 uint8_t temp_osr;
812 uint8_t pres_osr;
813 uint8_t hum_osr;
814
815 /*Array to map OSR config register value to actual OSR */
816 uint8_t osr_sett_to_act_osr[] = {0, 1, 2, 4, 8, 16};
817
818 /* Mapping osr settings to the actual osr values e.g. 0b101 -> osr X16 */
819 if (settings->osr_t <= 5) {
820 temp_osr = osr_sett_to_act_osr[settings->osr_t];
821 } else {
822 temp_osr = 16;
823 }
824
825 if (settings->osr_p <= 5) {
826 pres_osr = osr_sett_to_act_osr[settings->osr_p];
827 } else {
828 pres_osr = 16;
829 }
830
831 if (settings->osr_h <= 5) {
832 hum_osr = osr_sett_to_act_osr[settings->osr_h];
833 } else {
834 hum_osr = 16;
835 }
836
837 max_delay = (uint32_t)((BME280_MEAS_OFFSET + (BME280_MEAS_DUR * temp_osr) +
838 ((BME280_MEAS_DUR * pres_osr) + BME280_PRES_HUM_MEAS_OFFSET) +
839 ((BME280_MEAS_DUR * hum_osr) + BME280_PRES_HUM_MEAS_OFFSET)) /
840 BME280_MEAS_SCALING_FACTOR);
841
842 return max_delay;
843}

868static int8_t set_osr_humidity_settings(const struct bme280_settings* settings,
869 struct bme280_dev* dev) {
870 int8_t rslt;
871 uint8_t ctrl_hum;
872 uint8_t ctrl_meas;
873 uint8_t reg_addr = BME280_CTRL_HUM_ADDR;
874
875 ctrl_hum = settings->osr_h & BME280_CTRL_HUM_MSK;
876
877 /* Write the humidity control value in the register */
878 rslt = bme280_set_regs(&reg_addr, &ctrl_hum, 1, dev);
879
880 /* Humidity related changes will be only effective after a
881 * write operation to ctrl_meas register
882 */
883 if (rslt == BME280_OK) {
884 reg_addr = BME280_CTRL_MEAS_ADDR;
885 rslt = bme280_get_regs(reg_addr, &ctrl_meas, 1, dev);
886
887 if (rslt == BME280_OK) {
888 rslt = bme280_set_regs(&reg_addr, &ctrl_meas, 1, dev);
889 }
890 }
891
892 return rslt;
