64int8_t i2c_write(uint8_t reg_addr, const uint8_t* reg_data, uint32_t length, void* intf_ptr) {
65 uint8_t* buf;
66
67 // Sensirion adds CRC using their own methods, so we'll assume the address is already added if it
68 // is null
69 uint8_t address_offset = reg_addr != 0 ? 1 : 0;
70
71 struct identifier id;
72 id = *((struct identifier*)intf_ptr);
73
74 buf = malloc(length + address_offset);
75
76 if (address_offset)
77 buf[0] = reg_addr;
78
79 memcpy(buf + address_offset, reg_data, length);
80
81 if (write(id.fd, buf, length + 1) < (uint16_t)length)
82 return -2;
83
84 free(buf);
85
86 return 0;
87}
