84int8_t sht3x_init(struct sht3x_sensor_data* sht, uint8_t addr) {
85 int8_t rslt = STATUS_OK;
86
87 if (configure_mux()) {
88 syslog(LOG_CRIT, "Failed to configure mux switching.");
89 exit(1);
90 }
91
92 int8_t* fd = addr == SHT3X_I2C_ADDR_DFLT ? &fd_44 : &fd_45;
93
94 if (i2c_open(fd, addr)) {
95 syslog(LOG_CRIT, "Failed to open bus");
96 exit(SENSOR_FAIL);
97 }
98
99 ioctl(*fd, 0x0703, addr);
100
101 sht->id.fd = *fd;
102 rslt = sht3x_probe(sht);
103
104 return rslt;
105}

128 int32_t temperature, humidity;

178int16_t sht3x_read_serial(struct sht3x_sensor_data* sht, uint32_t* serial) {
179 int16_t ret;
180 uint8_t serial_bytes[4];
181
182 ret = sensirion_i2c_write_cmd(sht, SHT3X_CMD_READ_SERIAL_ID);
183 delay_us(SHT3X_CMD_DURATION_USEC, NULL);
184
185 if (ret == STATUS_OK) {
186 ret = sensirion_i2c_read_words_as_bytes(sht, serial_bytes, SENSIRION_NUM_WORDS(serial_bytes));
187 *serial = sensirion_bytes_to_uint32_t(serial_bytes);
188 }
189 return ret;
190}

192int16_t sht3x_set_alert_thd(struct sht3x_sensor_data* sht,
193 sht3x_alert_thd_t thd,
194 uint32_t humidity,
195 int32_t temperature) {
196 int16_t ret;
197 uint16_t rawT;
198 uint16_t rawRH;
199 uint16_t limitVal = 0U;
200
201 temperature_to_tick(temperature, &rawT);
202 humidity_to_tick(humidity, &rawRH);
203
204 /* convert inputs to alert threshold word */
205 limitVal = (rawRH & SHT3X_HUMIDITY_LIMIT_MSK);
206 limitVal |= ((rawT >> 7) & SHT3X_TEMPERATURE_LIMIT_MSK);
207
208 switch (thd) {
209 case SHT3X_HIALRT_SET:
210 ret = sensirion_i2c_write_cmd_with_args(sht, SHT3X_CMD_WRITE_HIALRT_LIM_SET, &limitVal, 1);
211 break;
212
213 case SHT3X_HIALRT_CLR:
214 ret = sensirion_i2c_write_cmd_with_args(sht, SHT3X_CMD_WRITE_HIALRT_LIM_CLR, &limitVal, 1);
215 break;
216
217 case SHT3X_LOALRT_CLR:
218 ret = sensirion_i2c_write_cmd_with_args(sht, SHT3X_CMD_WRITE_LOALRT_LIM_CLR, &limitVal, 1);
219 break;
220
221 case SHT3X_LOALRT_SET:
222 ret = sensirion_i2c_write_cmd_with_args(sht, SHT3X_CMD_WRITE_LOALRT_LIM_SET, &limitVal, 1);
223 break;
224
225 default:
226 ret = STATUS_ERR_INVALID_PARAMS;
227 break;
228 }
229 return ret;
230}

232int16_t sht3x_get_alert_thd(struct sht3x_sensor_data* sht,
233 sht3x_alert_thd_t thd,
234 int32_t* humidity,
235 int32_t* temperature) {
236 int16_t ret;
237 uint16_t word;
238 uint16_t rawT;
239 uint16_t rawRH;
240
241 switch (thd) {
242 case SHT3X_HIALRT_SET:
243 ret = sensirion_i2c_read_cmd(sht, SHT3X_CMD_READ_HIALRT_LIM_SET, &word, 1);
244 break;
245
246 case SHT3X_HIALRT_CLR:
247 ret = sensirion_i2c_read_cmd(sht, SHT3X_CMD_READ_HIALRT_LIM_CLR, &word, 1);
248 break;
249
250 case SHT3X_LOALRT_CLR:
251 ret = sensirion_i2c_read_cmd(sht, SHT3X_CMD_READ_LOALRT_LIM_CLR, &word, 1);
252 break;
253
254 case SHT3X_LOALRT_SET:
255 ret = sensirion_i2c_read_cmd(sht, SHT3X_CMD_READ_LOALRT_LIM_SET, &word, 1);
256 break;
257
258 default:
259 ret = STATUS_ERR_INVALID_PARAMS;
260 break;
261 }
262
263 /* convert threshold word to alert settings in 10*%RH & 10*°C */
264 rawRH = (word & SHT3X_HUMIDITY_LIMIT_MSK);
265 rawT = ((word & SHT3X_TEMPERATURE_LIMIT_MSK) << 7);
266
267 tick_to_humidity(rawRH, humidity);
268 tick_to_temperature(rawT, temperature);
269
270 return ret;
271}

273void tick_to_temperature(uint16_t tick, int32_t* temperature) {
274 *temperature = ((21875 * (int32_t)tick) >> 13) - 45000;
275}

277void tick_to_humidity(uint16_t tick, int32_t* humidity) {
278 *humidity = ((12500 * (int32_t)tick) >> 13);
279}

281void temperature_to_tick(int32_t temperature, uint16_t* tick) {
282 *tick = (uint16_t)((temperature * 12271 + 552195000) >> 15);
283}

285void humidity_to_tick(int32_t humidity, uint16_t* tick) {
286 *tick = (uint16_t)((humidity * 21474) >> 15);
