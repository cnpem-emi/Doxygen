256int8_t bmp2_get_regs(uint8_t reg_addr, uint8_t* reg_data, uint32_t len, struct bmp2_dev* dev) {
257 int8_t rslt;
258
259 rslt = null_ptr_check(dev);
260
261 if ((rslt == BMP2_OK) && (reg_data != NULL)) {
262 /* Mask the register address' MSB if interface selected is SPI */
263 if (dev->intf == BMP2_SPI_INTF) {
264 reg_addr = reg_addr | BMP2_SPI_RD_MASK;
265 }
266
267 dev->intf_rslt = dev->read(reg_addr, reg_data, len, dev->intf_ptr);
268
269 /* Check for communication error and mask with an internal error code */
270 if (dev->intf_rslt != BMP2_INTF_RET_SUCCESS) {
271 rslt = BMP2_E_COM_FAIL;
272 }
273 } else {
274 rslt = BMP2_E_NULL_PTR;
275 }
276
277 return rslt;

284int8_t bmp2_set_regs(uint8_t* reg_addr,
285 const uint8_t* reg_data,
286 uint32_t len,
287 struct bmp2_dev* dev) {
288 int8_t rslt;
289 uint8_t temp_buff[8]; /* Typically not to write more than 4 registers */
290 uint32_t temp_len;
291 uint8_t reg_addr_cnt;
292
293 if (len > BMP2_MAX_LEN) {
294 len = BMP2_MAX_LEN;
295 }
296
297 rslt = null_ptr_check(dev);
298
299 if ((rslt == BMP2_OK) && (reg_addr != NULL) && (reg_data != NULL)) {
300 if (len > 0) {
301 temp_buff[0] = reg_data[0];
302
303 /* Mask the register address' MSB if interface selected is SPI */
304 if (dev->intf == BMP2_SPI_INTF) {
305 /* Converting all the reg address into proper SPI write address
306 * i.e making MSB(R/`W) bit 0
307 */
308 for (reg_addr_cnt = 0; reg_addr_cnt < len; reg_addr_cnt++) {
309 reg_addr[reg_addr_cnt] = reg_addr[reg_addr_cnt] & BMP2_SPI_WR_MASK;
310 }
311 }
312
313 /* Burst write mode */
314 if (len > 1) {
315 /* Interleave register address w.r.t data for burst write */
316 interleave_data(reg_addr, temp_buff, reg_data, len);
317 temp_len = ((len * 2) - 1);
318 } else {
319 temp_len = len;
320 }
321
322 dev->intf_rslt = dev->write(reg_addr[0], temp_buff, temp_len, dev->intf_ptr);
323
324 /* Check for communication error and mask with an internal error code */
325 if (dev->intf_rslt != BMP2_INTF_RET_SUCCESS) {
326 rslt = BMP2_E_COM_FAIL;
327 }
328 } else {
329 rslt = BMP2_E_INVALID_LEN;
330 }
331 } else {
332 rslt = BMP2_E_NULL_PTR;
333 }
334
335 return rslt;
336}

608static int8_t conf_sensor(uint8_t mode, const struct bmp2_config* conf, struct bmp2_dev* dev) {
609 int8_t rslt;
610 uint8_t temp[2] = {0, 0};
611 uint8_t reg_addr[2] = {BMP2_REG_CTRL_MEAS, BMP2_REG_CONFIG};
612
613 if (conf != NULL) {
614 rslt = bmp2_get_regs(BMP2_REG_CTRL_MEAS, temp, 2, dev);
615
616 if (rslt == BMP2_OK) {
617 /* Here the intention is to put the device to sleep
618 * within the shortest period of time
619 */
620 rslt = bmp2_soft_reset(dev);
621
622 if (rslt == BMP2_OK) {
623 set_os_mode(temp, conf);
624 temp[1] = BMP2_SET_BITS(temp[1], BMP2_STANDBY_DURN, conf->odr);
625 temp[1] = BMP2_SET_BITS(temp[1], BMP2_FILTER, conf->filter);
626 temp[1] = BMP2_SET_BITS_POS_0(temp[1], BMP2_SPI3_ENABLE, conf->spi3w_en);
627
628 rslt = bmp2_set_regs(reg_addr, temp, 2, dev);
629
630 if ((rslt == BMP2_OK) && (mode != BMP2_POWERMODE_SLEEP)) {
631 dev->power_mode = mode;
632
633 /* Write only the power mode register in a separate write */
634 temp[0] = BMP2_SET_BITS_POS_0(temp[0], BMP2_POWERMODE, mode);
635 rslt = bmp2_set_regs(reg_addr, temp, 1, dev);
636 }
637 }
638 }
639 } else {
640 rslt = BMP2_E_NULL_PTR;
641 }
642
643 return rslt;
644}

650static void set_os_mode(uint8_t* reg_data, const struct bmp2_config* conf) {
651 switch (conf->os_mode) {
652 case BMP2_OS_MODE_ULTRA_LOW_POWER:
653 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_TEMP, BMP2_OS_1X);
654 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_PRES, BMP2_OS_1X);
655 break;
656 case BMP2_OS_MODE_LOW_POWER:
657 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_TEMP, BMP2_OS_1X);
658 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_PRES, BMP2_OS_2X);
659 break;
660 case BMP2_OS_MODE_STANDARD_RESOLUTION:
661 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_TEMP, BMP2_OS_1X);
662 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_PRES, BMP2_OS_4X);
663 break;
664 case BMP2_OS_MODE_HIGH_RESOLUTION:
665 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_TEMP, BMP2_OS_1X);
666 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_PRES, BMP2_OS_8X);
667 break;
668 case BMP2_OS_MODE_ULTRA_HIGH_RESOLUTION:
669 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_TEMP, BMP2_OS_2X);
670 reg_data[0] = BMP2_SET_BITS(reg_data[0], BMP2_OS_PRES, BMP2_OS_16X);
671 break;
672 default:
673 break;
674 }
675}

681static int8_t parse_sensor_data(const uint8_t* reg_data, struct bmp2_uncomp_data* uncomp_data) {
682 int8_t rslt;
683
684 /* Variables to store the sensor data */
685 uint32_t data_xlsb;
686 uint32_t data_lsb;
687 uint32_t data_msb;
688
689 /* Store the parsed register values for pressure data */
690 data_msb = (uint32_t)reg_data[0] << 12;
691 data_lsb = (uint32_t)reg_data[1] << 4;
692 data_xlsb = (uint32_t)reg_data[2] >> 4;
693 uncomp_data->pressure = data_msb | data_lsb | data_xlsb;
694
695 /* Store the parsed register values for temperature data */
696 data_msb = (int32_t)reg_data[3] << 12;
697 data_lsb = (int32_t)reg_data[4] << 4;
698 data_xlsb = (int32_t)reg_data[5] >> 4;
699 uncomp_data->temperature = (int32_t)(data_msb | data_lsb | data_xlsb);
700
701 rslt = st_check_boundaries((int32_t)uncomp_data->temperature, (int32_t)uncomp_data->pressure);
702
703 return rslt;
704}

715 int8_t rslt = BMP2_OK;
716 double var1, var2;

718

720 (((double)uncomp_data->temperature) / 16384.0 - ((double)dev->calib_param.dig_t1) / 1024.0) *
721 ((double)dev->calib_param.dig_t2);
722 var2 = ((((double)uncomp_data->temperature) / 131072.0 -
723 ((double)dev->calib_param.dig_t1) / 8192.0) *
724 (((double)uncomp_data->temperature) / 131072.0 -
725 ((double)dev->calib_param.dig_t1) / 8192.0)) *
726 ((double)dev->calib_param.dig_t3);
727
728 dev->calib_param.t_fine = (int32_t)(var1 + var2);

730
731 if (temperature < BMP2_MIN_TEMP_DOUBLE) {
732 temperature = BMP2_MIN_TEMP_DOUBLE;
733 rslt = BMP2_W_MIN_TEMP;

751 const struct bmp2_uncomp_data* uncomp_data,
752 const struct bmp2_dev* dev) {
753 int8_t rslt = BMP2_OK;
754 double var1, var2;
755 double pressure = 0.0;
756
757 var1 = ((double)dev->calib_param.t_fine / 2.0) - 64000.0;
758 var2 = var1 * var1 * ((double)dev->calib_param.dig_p6) / 32768.0;
759 var2 = var2 + var1 * ((double)dev->calib_param.dig_p5) * 2.0;
760 var2 = (var2 / 4.0) + (((double)dev->calib_param.dig_p4) * 65536.0);
761 var1 = (((double)dev->calib_param.dig_p3) * var1 * var1 / 524288.0 +
762 ((double)dev->calib_param.dig_p2) * var1) /
763 524288.0;
764 var1 = (1.0 + var1 / 32768.0) * ((double)dev->calib_param.dig_p1);
765
766 if (var1 < 0 || var1 > 0) {
767 pressure = 1048576.0 - (double)uncomp_data->pressure;
768 pressure = (pressure - (var2 / 4096.0)) * 6250.0 / var1;
769 var1 = ((double)dev->calib_param.dig_p9) * pressure * pressure / 2147483648.0;
770 var2 = pressure * ((double)dev->calib_param.dig_p8) / 32768.0;

772 pressure = pressure + (var1 + var2 + ((double)dev->calib_param.dig_p7)) / 16.0;

775 pressure = BMP2_MIN_PRES_DOUBLE;
